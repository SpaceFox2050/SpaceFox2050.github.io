# Triangle Times
```
int a, b, c;
    cin >> a >> b >> c;
    if(a == 60 && b == 60 && c  == 60){
        cout << "Equilateral";
        return;
    }else if(a + b + c == 180){
        if(a == b || a == c || b  == c){
            cout << "Isosceles";
            return;
        }else{
            cout << "Scalene";
            return;
        }
    }else{
        cout << "Error";
    }
```
